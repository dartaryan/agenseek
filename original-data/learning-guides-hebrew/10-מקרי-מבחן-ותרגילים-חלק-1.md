// controllers/userController.js
const User = require('../models/User');
const { isValidObjectId } = require('mongoose');

exports.updateUserStatus = async (req, res) => {
  try {
    const { id } = req.params;
    const { status } = req.body;

    // Validation
    if (!isValidObjectId(id)) {
      return res.status(400).json({
        success: false,
        message: 'Invalid user ID format'
      });
    }

    if (!status || !['active', 'inactive'].includes(status)) {
      return res.status(400).json({
        success: false,
        message: 'Status must be either "active" or "inactive"'
      });
    }

    // Find and update user
    const user = await User.findById(id);
    
    if (!user) {
      return res.status(404).json({
        success: false,
        message: 'User not found'
      });
    }

    user.status = status;
    await user.save();

    // Log the change
    console.log(`User status updated: ${user.email} â†’ ${status} by ${req.user.email}`);

    return res.status(200).json({
      success: true,
      message: 'User status updated successfully',
      user: {
        id: user._id,
        email: user.email,
        name: user.name,
        status: user.status
      }
    });

  } catch (error) {
    console.error('Error updating user status:', error);
    return res.status(500).json({
      success: false,
      message: 'Server error while updating user status'
    });
  }
};

import axios from 'axios';

const API_URL = '/api/categories';

export const categoriesApi = {
  getAll: async () => {
    const response = await axios.get(API_URL);
    return response.data;
  },

  create: async (category) => {
    const response = await axios.post(API_URL, category);
    return response.data;
  },

  update: async (id, category) => {
    const response = await axios.put(`${API_URL}/${id}`, category);
    return response.data;
  },

  delete: async (id) => {
    const response = await axios.delete(`${API_URL}/${id}`);
    return response.data;
  }
};

// src/services/orderService.js
export function calculateOrderTotal(items, discountCode = null, shippingMethod = 'standard') {
  // Calculate subtotal
  let subtotal = items.reduce((sum, item) => {
    return sum + (item.price * item.quantity);
  }, 0);

  // Apply discount
  let discount = 0;
  if (discountCode) {
    if (discountCode === 'SAVE10') {
      discount = subtotal * 0.1;
    } else if (discountCode === 'SAVE20') {
      discount = subtotal * 0.2;
    } else if (discountCode === 'FREESHIP') {
      discount = 0; // Free shipping handled separately
    }
  }

  // Calculate shipping
  let shipping = 0;
  if (discountCode !== 'FREESHIP') {
    if (shippingMethod === 'standard') {
      shipping = subtotal > 100 ? 0 : 10;
    } else if (shippingMethod === 'express') {
      shipping = 20;
    }
  }

  const total = subtotal - discount + shipping;
  
  return {
    subtotal,
    discount,
    shipping,
    total
  };
}

### POST /api/contacts/bulk-import

Import multiple contacts at once.

**Authentication:** Required (OAuth 2.0)

**Request:**
```json
{
  "contacts": [
    {
      "firstName": "John",
      "lastName": "Doe",
      "email": "john@example.com",
      "phone": "+1234567890"
    },
    {
      "firstName": "Jane",
      "lastName": "Smith",
      "email": "jane@example.com"
    }
  ],
  "skipDuplicates": true
}
```

**Response (201 Created):**
```json
{
  "success": true,
  "imported": 2,
  "skipped": 0,
  "contacts": [...]
}
```

**Errors:**
- `400` - Invalid contact data
- `401` - Unauthorized
- `413` - Too many contacts (max 1000)

## Authentication

Our API uses **OAuth 2.0** for authentication. 

### Getting Started

1. **Register your application** at https://app.ourcrm.com/developers
2. **Get your credentials:** Client ID and Client Secret
3. **Obtain an access token** using the OAuth 2.0 flow

### OAuth 2.0 Flow

#### Step 1: Authorization Request

Redirect the user to:
```
GET https://app.ourcrm.com/oauth/authorize
  ?client_id=YOUR_CLIENT_ID
  &redirect_uri=YOUR_REDIRECT_URI
  &response_type=code
  &scope=contacts:read contacts:write
```

#### Step 2: Exchange Code for Token

```bash
POST https://app.ourcrm.com/oauth/token
Content-Type: application/x-www-form-urlencoded

grant_type=authorization_code
&code=AUTH_CODE
&client_id=YOUR_CLIENT_ID
&client_secret=YOUR_CLIENT_SECRET
&redirect_uri=YOUR_REDIRECT_URI
```

**Response:**
```json
{
  "access_token": "eyJhbGc...",
  "token_type": "Bearer",
  "expires_in": 3600,
  "refresh_token": "def50200..."
}
```

#### Step 3: Use Access Token

Include the token in all API requests:
```bash
GET https://api.ourcrm.com/v2/contacts
Authorization: Bearer YOUR_ACCESS_TOKEN
```

### Token Refresh

When your token expires, use the refresh token:
```bash
POST https://app.ourcrm.com/oauth/token
Content-Type: application/x-www-form-urlencoded

grant_type=refresh_token
&refresh_token=YOUR_REFRESH_TOKEN
&client_id=YOUR_CLIENT_ID
&client_secret=YOUR_CLIENT_SECRET
```

// examples/auth-example.js - ×§×•×‘×¥ ×—×“×©

/**
 * OAuth 2.0 Authentication Example
 * 
 * This example shows how to authenticate with our API using OAuth 2.0
 */

const axios = require('axios');

const CLIENT_ID = 'your_client_id';
const CLIENT_SECRET = 'your_client_secret';
const REDIRECT_URI = 'http://localhost:3000/callback';

// Step 1: Get authorization code (user clicks on this URL)
const authUrl = `https://app.ourcrm.com/oauth/authorize?client_id=${CLIENT_ID}&redirect_uri=${REDIRECT_URI}&response_type=code&scope=contacts:read contacts:write`;
console.log('Visit this URL to authorize:', authUrl);

// Step 2: Exchange code for token
async function getAccessToken(authCode) {
  try {
    const response = await axios.post('https://app.ourcrm.com/oauth/token', {
      grant_type: 'authorization_code',
      code: authCode,
      client_id: CLIENT_ID,
      client_secret: CLIENT_SECRET,
      redirect_uri: REDIRECT_URI
    });

    return response.data;
  } catch (error) {
    console.error('Error getting access token:', error.response.data);
    throw error;
  }
}

// Step 3: Use the access token
async function getContacts(accessToken) {
  try {
    const response = await axios.get('https://api.ourcrm.com/v2/contacts', {
      headers: {
        'Authorization': `Bearer ${accessToken}`
      }
    });

    return response.data;
  } catch (error) {
    console.error('Error fetching contacts:', error.response.data);
    throw error;
  }
}

// Step 4: Refresh token when expired
async function refreshAccessToken(refreshToken) {
  try {
    const response = await axios.post('https://app.ourcrm.com/oauth/token', {
      grant_type: 'refresh_token',
      refresh_token: refreshToken,
      client_id: CLIENT_ID,
      client_secret: CLIENT_SECRET
    });

    return response.data;
  } catch (error) {
    console.error('Error refreshing token:', error.response.data);
    throw error;
  }
}

// Example usage
async function main() {
  // After user authorizes, you'll get the code from redirect
  const authCode = 'code_from_redirect';
  
  // Get initial tokens
  const tokens = await getAccessToken(authCode);
  console.log('Access token:', tokens.access_token);
  
  // Use the API
  const contacts = await getContacts(tokens.access_token);
  console.log('Contacts:', contacts);
  
  // When token expires, refresh it
  const newTokens = await refreshAccessToken(tokens.refresh_token);
  console.log('New access token:', newTokens.access_token);
}

module.exports = { getAccessToken, getContacts, refreshAccessToken };
